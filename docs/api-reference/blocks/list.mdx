title: List Blocks
api: GET /blocks
description: "Retrieve study blocks with optional filters."
---

<ParamField query="date" type="string">
  Filter by specific date (`YYYY-MM-DD`)
</ParamField>

<ParamField query="date_from" type="string">
  Start of date range (`YYYY-MM-DD`)
</ParamField>

<ParamField query="date_to" type="string">
  End of date range (`YYYY-MM-DD`)
</ParamField>

<ParamField query="subject_id" type="string">
  Filter by activity/subject ID
</ParamField>

<ParamField query="done" type="boolean">
  Filter by completion status
</ParamField>

<ParamField query="limit" type="number" default="50">
  Max items per page (1-100)
</ParamField>
